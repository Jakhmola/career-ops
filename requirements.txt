#

# Arch/Debian where system pip is blocked):

#   python -m venv .venv
#   .venv/bin/pip install -r requirements.txt      (Python 3.10+)
#
# providers/jobspy.mjs auto-detects ./.venv (or honors $JOBSPY_PYTHON), so after

# errors.
